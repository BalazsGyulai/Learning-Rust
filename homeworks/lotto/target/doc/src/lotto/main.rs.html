115/** 
116Converts a string into a Vector in which each element is character from the string represented as a u8 value. 
117E.g. From String "12" it will create a Vector that contains [49, 50] and the function will return this Vector
118*/
119fn string_to_vector(input: String) -> Vec<u8> {
120    // input_in_bytes stores an array &[u8]
121    let input_in_bytes = input.as_bytes();
122    // my_vector will be our vector in which we copy the elements of input_in_bytes.
123    let mut my_vector: Vec<u8> = Vec::with_capacity(input_in_bytes.len());
124
125    for i in 0..input_in_bytes.len() {
126        my_vector.push(input_in_bytes[i] as u8);
127    }
128
129    my_vector
130}

132/** 
133Gets a vector as an argument. Since we don't care about the characters that are not numbers, we will make every unecessary character into a space that is 32 in the ASCII table. After we did this, we will remove any duplicate spaces.
134*/
135fn remove_extra_characters(vector: &mut Vec<u8>) {
136    /*! This function changes the unecessary characters into spaces that are represented as '32' in the ASCII table. */
137    change_character_to_space(vector);
138
139    let mut i: u64 = 0;
140
141    /* 
142    This part removes the extra spaces in a way so only one space will be stay behind. This is necessary because we will use the spaces 
143    to know where a new digit starts or ends.
144    */
145    while i < (vector.len() as u64) - 1 {
146        if vector[i as usize] == 32 && vector[(i + 1) as usize] == 32 {
147            // custom_remove is a function that removes the i element from an array.
148            *vector = custom_remove(vector, (i + 1) as usize);
149        } else {
150            i += 1;
151        }
152    }
153
154    // In case there is an extra space in the front it will remove it from the front of the array
155    if vector[0] == 32 {
156        *vector = custom_remove(vector, 0);
157    }
158}

160/// Changes every character into a space represented as '32' in the ASCII table. We know that 0 -> 48 and 9 -> 57, every other number is not important for us
161fn change_character_to_space(vector: &mut Vec<u8>) {
162    for i in 0..vector.len() as usize {
163        if vector[i as usize] < 48 || vector[i as usize] > 57 {
164            vector[i as usize] = 32;
165        }
166    }
167}

169/// Removes the i element of a given array, then it returns the new array.
170fn custom_remove(array: &Vec<u8>, i: usize) -> Vec<u8> {
171    // we declare a Vector that has the capacity of holding all the elements of the given array minus one element becuase we will remove one.
172    let mut new_array: Vec<u8> = Vec::with_capacity(array.len() - 1);
173
174    for y in 0..array.len() {
175        // we add every element to the new_array but the element at the index of i
176        if (y as usize) != i {
177            new_array.push(array[y as usize]);
178        }
179    }
180
181    new_array
182}

184/** 
185Gets an argument (input) that is in bytes. It will search for the first space (32), will store where it is then will convert back the bytes back into integers and repeats this process on the whole array. Finally, will return the array that holds the integers.
186*/
187fn convert_input_back(input: &mut Vec<u8>) -> Vec<i64> {
188    // we will store here the integers that are converted back.
189    let mut new_vector: Vec<i64> = Vec::new();
190    // stores where the array was cut the last time.
191    let mut cut_front: u64 = 0;
192
193    for i in 0..input.len() {
194        // goes until finds a space
195        if input[i as usize] == 32 {
196            // adds the integer that is converted back
197            // cut_front -> the last place the array was cut, default is 0
198            // i -> place until it needs to be cut (it is not counted in when cutting it)
199            // input -> the array which we work on
200            new_vector.push(get_value_back(cut_front, i as u64, input));
201
202            // stores the place where the array was cut. (i + 1) because we don't need to convert back the space.
203            if i + 1 < input.len() {
204                cut_front = (i + 1) as u64;
205            }
206        }
207    }
208
209    new_vector
210}

212/// Converts a particular part of an array back to integers.
213fn get_value_back(start: u64, end: u64, array: &Vec<u8>) -> i64 {
214    // stores the integer
215    let mut sum: i64 = 0;
216
217    // sees only a particular part of the array
218    for i in start..end {
219        // we make a multiply variable that hold 1, 10, 100, ..., this is that knows where the number takes place.
220        // E.g:
221        // We have the number 123 it can be represented as
222        // 1 * 100 + 2 * 10 + 3 * 1
223        let multiply: i64 = {
224            let mut number: i64 = 1;
225
226            for _ in i..end - 1 {
227                number *= 10;
228            }
229
230            number
231        };
232
233        // custom parse gets a byte number and gets back an i64 number, then we multiply it so we get back the origin number
234        sum += custom_parse(array[i as usize]) * multiply;
235    }
236
237    sum
238}

243    match character {

253        _ => 9,

297/// Gets an array and arranges its elements in ascending order
298fn arrange_array_in_ascend(array: &mut Vec<i64>) {
299    for i in 0..array.len() {
300        for y in i..array.len() {
301            if array[i as usize] > array[y as usize] {
302                let temp: i64 = array[y as usize];
303                array[y as usize] = array[i as usize];
304                array[i as usize] = temp;
305            }
306        }
307    }
308}

377/// Handles checking how many numbers are the same from the user input inside the numbers generated by the computer
378fn determining_rank(input: &Vec<i64>, computer_generated_numbers: &Vec<i64>, bonus_number: &i64) {
379    let mut normal_match: u64 = 0; // counter of the same numbers
380    let mut bonus_match: bool = false; // will be true when a user number is the same as the bonus number
381
382    // checks all the numbers of the user 
383    for i in 0..input.len() {
384        let mut found_match: bool = false; // has to reset everytime since we compare a new number with the elements of the computer
385        let mut y = 0; // has to reset because we will start comparing the first element of the computer
386
387        // runs until it finds numbers that are the same, or reaches the end of the array of the computer
388        while y < computer_generated_numbers.len() && !found_match {
389            if input[i as usize] == computer_generated_numbers[y as usize] {
390                normal_match += 1;
391                found_match = true;
392            }
393
394            y += 1;
395        }
396
397        // also compares the user element with the bonus number
398        if !bonus_match && input[i as usize] == *bonus_number {
399            bonus_match = true;
400        }
401    }
402
403    // this is the part that is the table from the task description
404    match (normal_match, bonus_match) {
405        (6, false) => print_prize_and_correct_guess(1, normal_match),
406        (5, true) => print_prize_and_correct_guess(2, normal_match),
407        (5, false) => print_prize_and_correct_guess(3, normal_match),
408        (4, true) => print_prize_and_correct_guess(4, normal_match),
409        (4, false) => print_prize_and_correct_guess(5, normal_match),
410        (3, true) => print_prize_and_correct_guess(6, normal_match),
411        (3, false) => print_prize_and_correct_guess(7, normal_match),
412        _ => print_prize_and_correct_guess(8, normal_match),
413    }
414}
